class Solution {
    public int uniquePaths(int m, int n) {
         //动态规划
        int[][] grid=new int[m][n];
        //第一列的数据都是1
        for(int i=0;i<m;i++){
            grid[i][0]=1;
        }
        //第一行的数据都是1
        for(int i=0;i<n;i++){
            grid[0][i]=1;
        }
        //二维数组中的值是左边值加上上面值
        for(int i=1;i<m;i++){
            for(int j=1;j<n;j++){
                grid[i][j]=grid[i][j-1]+grid[i-1][j];
            }
        }
        return grid[m-1][n-1];
    }
}
